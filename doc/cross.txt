*Setting Example*

" cross | neovim/init.vim

" using vim-plug.
if filereadable(expand('~/.neovim/plugged/cross/plugins/command.vim'))
  source ~/.neovim/plugged/cross/plugins/command.vim
endif

" Added in vim-plug
Plug 'mfussenegger/nvim-dap'
Plug 'rcarriga/nvim-dap-ui'
Plug 'takkii/cross'

" Set before appearance etc.

lua require('cross').setup()

# Manually edit the settings for rdbg etc.
lua require('cross').setup {
  -- Example, rdbg configurations. The path is connected separately.
  rdbg = {
    initialize_timeout_sec = 20,
    port = "${port}",
    args = {},
    build_flags = "",
    detached = true,
    cwd = nil,
  },
  -- Use anyenv, Other Paths, / rails / bundle / readapt / rackup / rspec / rake / 
  rails = {
    home = os.getenv("HOME"),
    path = home .. '/.anyenv/envs/rbenv/shims/rails',
    -- rails = home .. '/.rbenv/shims/rails'
    -- rails = '/usr/local/bin/rails'
    -- rails = 'rails.bat'
    -- ... 
  },
}

lua require('dapui').setup()

:BreakPoint

:UIOpen

:Debug

" UI hidden
:UIToggle

" Release debug UI
:UIClose

:RPOpen
